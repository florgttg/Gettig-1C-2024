133uint16_t ConvertirVoltajeATemperatura(uint16_t voltaje)
134{
135 return TEMPERATURA_MIN + (voltaje - VOLTAJE_MIN) * (TEMPERATURA_MAX - VOLTAJE_MIN) / (VOLTAJE_MAX - VOLTAJE_MIN);

141void Task_Suministrar_Agua(void *pvParameter)
142{
143 uint16_t distancia = 0;
144 uint8_t tiempo_entrega = 0;
145 uint8_t tiempo_espera = 0;
146
147 while (1)
148 {
149 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
150 if (start)
151 {
152 distancia = HcSr04ReadDistanceInCentimeters();
153 if (distancia > 5 && distancia < 10)
154 {
155 if (tiempo_entrega < 5)
156 {
157 DispensarAgua();
158 if (tiempo_entrega == 0)
159 {
160 MostrarMensaje(1);
161 }
162 tiempo_entrega++;
163 }
164 else if (tiempo_espera < 5)
165 {
166 MostrarMensaje(2);
167 NoDispensarAgua();
168 tiempo_espera++;
169 }
170 else
171 {
172 tiempo_entrega = 0;
173 tiempo_espera = 0;
174 }
175 }
176 else
177 {
178 NoDispensarAgua();
179 }
180 }
181 }
182}

188void Task_Controlar_Temperatura(void *pvParameter)
189{
190 uint16_t temperaturamV;
191 uint16_t temperatura;
192 while (1)
193 {
194 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
195 if (start)
196 {
197 AnalogInputReadSingle(GPIO_SENSOR_TEMPERATURA, &temperaturamV); // Lee el valor de la temepratura en voltaje
198 temperatura = ConvertirVoltajeATemperatura(temperaturamV);
199 if (temperatura < TEMPERATURA_OBJETIVO)
200 {
201 CalentarAgua();
202 MostrarMensaje(2);
203 }
204 else
205 {
206 NoCalentarAgua();
207 MostrarMensaje(1);
208 }
209 }
210 }

230void app_main(void)
231{
232 HcSr04Init(GPIO_3, GPIO_2);
233 // Para la bomba

238 .input = GPIO_SENSOR_TEMPERATURA,
239 .mode = ADC_SINGLE,
240 .func_p = NULL,
241 .param_p = NULL,
242 .sample_frec = NULL};
243
244 AnalogInputInit(&senal_temperatura);
245 // Para la resitencia
246 GPIOInit(GPIO_RESISTENCIA, GPIO_OUTPUT);
247 GPIOOff(GPIO_RESISTENCIA);
248
249 // Para la uart
250 serial_config_t my_uart = {
251 .port = UART_PC,
252 .baud_rate = 115200,
253 .func_p = NULL,
254 .param_p = NULL
255 };
256 UartInit(&my_uart);
257
258 // ESTO CON INTERRUPCIONES
259 SwitchActivInt(SWITCH_1, &FunctionStart, NULL);
260 SwitchActivInt(SWITCH_2, &FunctionStop, NULL);
261
262 // Configuración del timer para control de suministro de agua
263 timer_config_t timer_suministro = {
264 .timer = TIMER_A,
265 .period = CONFIG_TIMER_SUMINISTRO,
266 .func_p = FunctionTimerA,
267 .param_p = NULL};
268 TimerInit(&timer_suministro);
269 // Configuración del timer para control de temperatura
270 timer_config_t timer_temperatura = {
271 .timer = TIMER_B,
272 .period = CONFIG_TIMER_TEMPERATURA,
273 .func_p = FunctionTimerB,
274 .param_p = NULL};
275 TimerInit(&timer_temperatura);
276
277 xTaskCreate(&Task_Suministrar_Agua, "Task_suministro", 2048, NULL, 5, &sumnistro_task_handle);
278 xTaskCreate(&Task_Controlar_Temperatura, "Task_temperatura", 2048, NULL, 5, &temperatura_task_handle);
279
280 TimerStart(timer_suministro.timer);
281 TimerStart(timer_temperatura.timer);
282}
