105void Task_Volar(void *pvParameter)
106{
107 float acel_x, acel_y, acel_z;
108 float angulo_anterior = 0;
109 float angulo; // angulo plano yz
110 float angulo_umbral_up = 8;
111 float angulo_umbral_down = -8;
112 float diferencia_angulos;
113 while (1)
114 {
115 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
116 if (start)
117 {
118 acel_x = ReadXValue();
119 acel_y = ReadYValue();
120 acel_z = ReadZValue();
121
122 float angulo = atan2(acel_y, acel_z) * 180 / M_PI; // Convertir a grados
123
124 //printf("El ángulo de inclinación en el plano YZ es: %.2f grados\n", angulo);
125
126 diferencia_angulos = angulo_anterior - angulo;
127 // printf("La diferencia de angulos es : %f grados\n", diferencia_angulos);
128
129 if ((diferencia_angulos) > angulo_umbral_up)
130 {
131 TeclaArriba();
132 angulo_anterior = angulo;
133 }
134 else if ((diferencia_angulos) < angulo_umbral_down)
135 {
136 TeclaAbajo();
137 angulo_anterior = angulo;
138 }
139 }
140 }

145void Task_Controles(void *pvParameter)
146{
147 // Chequea que el bluetooth este correctamente conectado con el led1
148 while (1)
149 {
150 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
151 switch (BleHidStatus())
152 {
153 case BLE_OFF:
154 LedOff(LED_BT);
155 break;
156 case BLE_DISCONNECTED:
157 LedToggle(LED_BT);
158 break;
159 case BLE_CONNECTED:
160 LedOn(LED_BT);
161 break;
162 }
163
164 if (start)
165 {
166 LedOn(LED_START);
167 }
168 else
169 {
170 LedOff(LED_START);
171 }
172 }
173}

204

207 SwitchActivInt(SWITCH_2, &TeclaBarra, NULL);
208
209 // Configuración del timer para control del vuelo del pajaro
210 timer_config_t timer_vuelo = {
211 .timer = TIMER_A,
212 .period = CONFIG_TIMER_VUELO,
213 .func_p = FunctionTimerA,
214 .param_p = NULL};
215 TimerInit(&timer_vuelo);
216
217 // Configuración del timer para control de bluetooth
218 timer_config_t timer_controles = {
219 .timer = TIMER_B,
220 .period = CONFIG_TIMER_CONTROLES,
221 .func_p = FunctionTimerB,
222 .param_p = NULL};
223 TimerInit(&timer_controles);
224
225 xTaskCreate(&Task_Volar, "Task_Volar", 4096, NULL, 5, &volar_task_handle);
226 xTaskCreate(&Task_Controles, "Task_control", 2048, NULL, 4, &controles_task_handle);
227
228 TimerStart(timer_vuelo.timer);
229 TimerStart(timer_controles.timer);
230}
