182void Task_Control_Humedad_y_Ph(void *pvParameter)
183{
184 uint16_t phMv;
185 uint16_t ph;
186 while (1)
187 {
188 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
189 if (start)
190 {
191 if (GPIORead(GPIO_SENSOR_HUMEDAD))
192 {
193 PrenderBomba(1);
194 }
195 else
196 ApagarBomba(1);
197 }
198 AnalogInputReadSingle(GPIO_SENSOR_PH, &phMv); // Lee el valor del ph en voltaje
199 ph = ConvertirVoltajeAPh(phMv);
200 if (ph < PH_BASICO)
201 {
202 PrenderBomba(3);
203 }
204 else
205 {
206 ApagarBomba(3);
207 }
208 if (ph > PH_ACIDO)
209 {
210 PrenderBomba(2);
211 }
212 else
213 {
214 ApagarBomba(2);
215 }
216 }
217}
